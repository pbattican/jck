block_annexation_mission_selection = {
	set_country_flag = { flag = open_annexation_event days = 15 value = 1 }
}

allow_annexation_mission_selection = {
	clr_country_flag = open_annexation_event
}

recheck_annexations = {
	allow_annexation_mission_selection = yes
	hidden_effect = {
		if = {
			limit = {
				NOT = { has_country_flag = KR_checking_for_possible_annexations }
			}
			set_country_flag = { flag = KR_checking_for_possible_annexations days = 7 value = 1 }
			country_event = { id = annex.1 days = 1 random_days = 3 }
		}
	}
}

recheck_annexations_immediate = {
	allow_annexation_mission_selection = yes
	hidden_effect = { country_event = annex.1 }
}
