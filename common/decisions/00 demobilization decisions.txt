demobilization = {
	demobilization_economic = {
		icon = revolt

		allowed = { always = no } #manually activated

		cancel_trigger = {
			OR = {
				has_unsupported_economic_law = no
				has_war = yes
			}
		}

		cost = 0
		is_good = no
		selectable_mission = yes
		days_mission_timeout = 60

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision complete demobilization_economic"
			if = {
				limit = { has_war_support < 0.15 }
				if = {
					limit = { has_idea = tot_economic_mobilisation }
					add_political_power = 300
					swap_ideas = {
						remove_idea = tot_economic_mobilisation
						add_idea = civilian_economy
					}
				}
				else_if = {
					limit = { has_idea = war_economy }
					add_political_power = 225
					swap_ideas = {
						remove_idea = war_economy
						add_idea = civilian_economy
					}
				}
				else_if = {
					limit = { has_idea = partial_economic_mobilisation }
					add_political_power = 150
					swap_ideas = {
						remove_idea = partial_economic_mobilisation
						add_idea = civilian_economy
					}
				}
				else_if = {
					limit = { has_idea = low_economic_mobilisation }
					add_political_power = 75
					swap_ideas = {
						remove_idea = low_economic_mobilisation
						add_idea = civilian_economy
					}
				}
			}
			else_if = {
				limit = { has_war_support < 0.25 }
				if = {
					limit = { has_idea = tot_economic_mobilisation }
					add_political_power = 225
					swap_ideas = {
						remove_idea = tot_economic_mobilisation
						add_idea = low_economic_mobilisation
					}
				}
				else_if = {
					limit = { has_idea = war_economy }
					add_political_power = 150
					swap_ideas = {
						remove_idea = war_economy
						add_idea = low_economic_mobilisation
					}
				}
				else_if = {
					limit = { has_idea = partial_economic_mobilisation }
					add_political_power = 75
					swap_ideas = {
						remove_idea = partial_economic_mobilisation
						add_idea = low_economic_mobilisation
					}
				}
			}
			else_if = {
				limit = {
					OR = {
						has_war_support < 0.5
						has_war = no
					}
				}
				if = {
					limit = { has_idea = tot_economic_mobilisation }
					add_political_power = 150
					swap_ideas = {
						remove_idea = tot_economic_mobilisation
						add_idea = partial_economic_mobilisation
					}
				}
				else_if = {
					limit = { has_idea = war_economy }
					add_political_power = 75
					swap_ideas = {
						remove_idea = war_economy
						add_idea = partial_economic_mobilisation
					}
				}
			}
			else_if = {
				limit = {
					has_war_support < 0.8
					has_idea = tot_economic_mobilisation
				}
				add_political_power = 75
				swap_ideas = {
					remove_idea = tot_economic_mobilisation
					add_idea = war_economy
				}
			}
		}

		timeout_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision timeout demobilization_economic"
			activate_mission = demobilization_economic
			if = {
				limit = { has_stability > 0 }
				add_stability = -0.05
			}
			add_political_power = -25
		}
	}

		timeout_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision timeout demobilization_manpower"

			}
		}
